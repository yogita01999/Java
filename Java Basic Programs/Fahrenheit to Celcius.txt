import java.util.*; 

class FahrenheitToCelcius{ 

    public static void main(String[] args) { 

        Scanner sc = new Scanner(System.in); 

        float temp; 

        System.out.println("Enter temperature in fahrenheit:"); 

        temp = sc.nextFloat(); 

         

        temp = (((temp -32) * 5) / 9); 

        System.out.println("Temperature in Celcius:" + temp); 

    } 

}
